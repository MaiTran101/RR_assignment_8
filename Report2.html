# URL of the Wikipedia page for the show 'Friends'
url = "https://en.wikipedia.org/wiki/List_of_Friends_episodes"

# Use pandas to scrape all tables from the Wikipedia page
tables = pd.read_html(url)
In[5]:
# Extract the first table
df = tables[0]

# Clean up and extract necessary columns
df.columns = [col[0] for col in df.columns]
df = df.drop(df.index[-1])
df[['Rating','Viewers (millions)']] = df[['Rating','Viewers (millions)']].replace({r'[^\d.]': ''}, regex=True).astype(float)
df['Season'] = df['Season'].replace({r'[^\d.]': ''}, regex=True).astype(int)

df_clean = df[['Season', 'Rating', 'Viewers (millions)']]
